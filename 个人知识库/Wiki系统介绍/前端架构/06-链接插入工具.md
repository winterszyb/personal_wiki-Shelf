# 前端架构 - 链接插入工具

## 功能概述

编辑器中提供图形化界面，用于插入指向其他 Wiki 文档的内部链接。

## 核心流程

```
用户点击"🔗 插入链接"
  → 弹出模态框
  → 显示目录树（只显示文件）
  → 用户选择目标文件
  → 输入链接显示文本（可选）
  → 点击"插入"
  → 计算相对路径
  → 在光标处插入 [文本](./相对路径.md)
```

## 核心函数

### 显示链接模态框

```javascript
async function showLinkModal() {
    document.getElementById('linkText').value = '';
    document.getElementById('linkModal').classList.add('active');
    selectedLinkPath = null;

    // 加载目录树
    const response = await fetch('/api/wiki/tree');
    const data = await response.json();
    document.getElementById('linkTreeContainer').innerHTML = 
        renderLinkTree(data.tree);
}
```

### 渲染链接选择树

```javascript
function renderLinkTree(nodes, level = 0) {
    return nodes.map(node => {
        const isFile = node.type === 'file';

        if (isFile) {
            return `
                <div class="link-tree-item"
                     onclick="selectLinkPath('${escapeJs(node.path)}', this)"
                     style="padding-left: ${12 + level * 20}px">
                    <span>📄</span> ${node.name}
                </div>
            `;
        } else {
            return `
                <div class="link-tree-item directory"
                     style="padding-left: ${12 + level * 20}px">
                    <span>📁</span> ${node.name}
                </div>
                <div class="link-tree-children">
                    ${renderLinkTree(node.children, level + 1)}
                </div>
            `;
        }
    }).join('');
}
```

### 选择目标文件

```javascript
let selectedLinkPath = null;

function selectLinkPath(path, element) {
    selectedLinkPath = path;
    document.querySelectorAll('.link-tree-item').forEach(el => {
        el.classList.remove('selected');
    });
    element.classList.add('selected');  // 紫色高亮
}
```

### 确认插入链接

```javascript
function confirmInsertLink() {
    if (!selectedLinkPath) {
        alert('请先选择一个目标文件');
        return;
    }

    const text = document.getElementById('linkText').value.trim();
    const linkText = text || selectedLinkPath.split('/').pop().replace('.md', '');

    // 计算相对路径
    const currentPath = currentFile || '';
    const currentDir = currentPath.lastIndexOf('/') > 0
        ? currentPath.substring(0, currentPath.lastIndexOf('/'))
        : '';

    const relativePath = calculateRelativePath(currentDir, selectedLinkPath);
    const markdownLink = `[${linkText}](./${relativePath})`;

    insertText(markdownLink);
    hideModal('linkModal');
}
```

### 相对路径计算

```javascript
function calculateRelativePath(fromDir, toPath) {
    if (!fromDir) return toPath;

    const fromParts = fromDir.split('/');
    const toParts = toPath.split('/');

    // 找到共同前缀长度
    let commonLength = 0;
    while (commonLength < fromParts.length &&
           commonLength < toParts.length &&
           fromParts[commonLength] === toParts[commonLength]) {
        commonLength++;
    }

    // 向上退的层级数
    const upCount = fromParts.length - commonLength;
    // 向下进入的路径
    const downParts = toParts.slice(commonLength);

    let result = '';
    for (let i = 0; i < upCount; i++) {
        result += '../';
    }
    result += downParts.join('/');

    return result;
}
```

## 示例

当前文件：`个人知识库/../../../clowder-ai架构wiki/01-项目结构.md`

目标文件：`个人知识库/../../../clowder-ai架构wiki/03-多智能体协作/多智能体协作机制.md`

计算过程：
```
fromDir = "个人知识库/../../../clowder-ai架构wiki"
toPath  = "个人知识库/../../../clowder-ai架构wiki/03-多智能体协作/多智能体协作机制.md"

共同前缀 = "个人知识库/../../../clowder-ai架构wiki" (3层)
向上退 = 0 层
downParts = ["03-多智能体协作", "多智能体协作机制.md"]

结果 = "03-多智能体协作/多智能体协作机制.md"
```

插入结果：
```markdown
[多智能体协作机制](./03-多智能体协作/多智能体协作机制.md)
```
