# 前端架构 - 目录树模块

## 核心函数

```javascript
function renderTree(nodes, level = 0) {
    return nodes.map((node, index) => {
        const icon = getIcon(node.name, node.type);
        const hasChildren = node.type === 'directory' && node.children && node.children.length > 0;
        const nodeId = `${node.path}-${index}`;
        const isExpanded = expandedNodes.has(nodeId);

        let html = `
            <div class="tree-node">
                <div class="tree-node-content"
                     data-path="${node.path}"
                     data-type="${node.type}"
                     data-id="${nodeId}"
                     onclick="handleNodeClick(...)"
                     oncontextmenu="handleContextMenu(...)">
                    <span class="tree-toggle ${hasChildren ? '' : 'hidden'} 
                          ${isExpanded ? 'expanded' : ''}">
                        ${hasChildren ? '▶' : ''}
                    </span>
                    <span class="tree-icon">${icon}</span>
                    <span class="tree-name">${node.name}</span>
                </div>
        `;

        if (hasChildren) {
            html += `<div class="tree-children ${isExpanded ? 'expanded' : ''}">`;
            html += renderTree(node.children, level + 1);
            html += `</div>`;
        }

        html += `</div>`;
        return html;
    }).join('');
}
```

## 交互功能

### 1. 展开/折叠

```javascript
function handleNodeClick(event, path, type, nodeId) {
    if (type === 'directory') {
        if (expandedNodes.has(nodeId)) {
            expandedNodes.delete(nodeId);  // 折叠
        } else {
            expandedNodes.add(nodeId);      // 展开
        }
        // 更新 UI：旋转箭头 + 显示/隐藏子节点
    } else {
        loadFile(path);  // 加载文件内容
    }
}
```

### 2. 选中高亮

```javascript
document.querySelectorAll('.tree-node-content').forEach(item => {
    item.classList.remove('active');
    if (item.dataset.path === filePath) {
        item.classList.add('active');  // 紫色渐变背景
    }
});
```

### 3. 右键菜单

```javascript
function handleContextMenu(event, path, type) {
    event.preventDefault();  // 阻止默认右键菜单
    const choice = prompt('选择操作:\n1 - 重命名\n2 - 删除\n其他 - 取消');
    if (choice === '1') showRenameModal(path);
    else if (choice === '2') showDeleteModal(path, type);
}
```
